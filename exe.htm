Inductive

Nat


    data nat: * :=
         (zero: () → nat)
         (succ: nat → nat)

List


    data list: (A:*) → * :=
         (nil: () → list A)
         (cons: A → list A → list A)

Vector


    data vector: (A:*) → nat → * :=
         (nil: () → vector A zero)
         (cons: (n: nat) → A → vector A n → vector A (succ n))

Logic

Unit


    data unit: *1 :=
         (star: () → unit)

Empty


    data empty: *1 := ()

True


    data true: Prop :=
         (intro: () → true)

False


    data false: Prop := ()

Eq


    data eq: (A:*) → A → A → Prop :=
         (refl: (x:A) → eq A x x)

Coinductive

Pure


  record pure: (P: * → *) → (A: *) → * :=
         (return: P A)

Functor


  record functor: (F: * → *) → (A B: *) → * :=
         (fmap: (A → B) → F A → F B)

Applicative


  record applicative: (F: * → *) → (A B: *) → *
  extend pure F A, functor F A B :=
         (ap: F (A → B) → F A → F B)

Monad


  record monad: (F: * → *) → (A B: *) → *
  extend pure F A, functor F A B :=
         (join: F (F A) → F B)
