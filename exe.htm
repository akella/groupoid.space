Syntax


     I := #identifier

     O := ∅ | O O | ( O ) |
          □ | O → O | 
          * | λ ( I : O ) → O |
          I | ∀ ( I : O ) → O |

     L := ∅ | L I
    
     A := ∅ | A ( L : O ) | A O

     F := ∅ | F ( I : O )

     E := ∅ | E <data|record> I : A := F

Inductive

Nat


    data nat: * :=
         (zero: () → nat)
         (succ: nat → nat)

List


    data list: (A:*) → * :=
         (nil: () → list A)
         (cons: A → list A → list A)

Vector


    data vector: (A:*) → nat → * :=
         (nil: () → vector A zero)
         (cons: (n: nat) → A → vector A n → vector A (succ n))

AST


    data om: * := 
         (nil: () → om)
         (star: () → om)
         (box: () → om)
         (var: name → om)
         (app: om → om → om)
         (arrow: name → om → om → om)
         (pi: name → om → om → om)

    data exe: * := 
         (record: name → om → om → exe)
         (data:   name → om → om → exe)

Coinductive

Pure


  record pure: (P: * → *) → (A: *) → * :=
         (return: P A)

Functor


  record functor: (F: * → *) → (A, B: *) → * :=
         (fmap: (A → B) → F A → F B)

Applicative


  record applicative: (F: * → *) → (A, B: *) → *
  extend pure F A, functor F A B :=
         (ap: F (A → B) → F A → F B)

Monad


  record monad: (F: * → *) → (A, B: *) → *
  extend pure F A, functor F A B :=
         (join: F (F A) → F B)

Logic

Unit


    data unit: *1 :=
         (star: () → unit)

Empty


    data empty: *1 := ()

True


    data True: Prop :=
         (intro: () → true)

False


    data False: Prop := ()

Eq


    data Eq: (A:*) → A → A → Prop :=
         (refl: (x:A) → Eq A x x)

Object


  record Ob: * :=
         (elem: *)
         (eq: Eq elem)
         (point: elem)

Homomorphism


  record Hom (X,Y: Ob): * :=
         (elem: X.elem → Y.elem)
         (eq: ∀ (x1, x2: X.elem) → X.eq x1 x2 → Y.eq (elem x1) (elem x2)) 
         (point: Y.eq (elem X.point) Y.point)

Setoid


  record Setoid: * :=
         (ob: *)
         (hom: ∀ (X,Y: ob) → Hom X Y)

Groupoid


  record Groupoid: * :=
         (ob: *)
         (hom: ∀ (a,b: ob) → (Hom a b) → Groupoid)
