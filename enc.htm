Impredicative Encoding

Categorical Semantics of List example

Initial Object


    let I = data List: (A:*) -> * :=
                 (nil: List A)
                 (cons: A -> List A -> List A)

Construct corresponding F-Algebra


    record ListAlg: (A:*) -> * :=
           (X: *)
           (nil: X)
           (cons: A -> X -> X)

Introduce List Morphisms


  record ListMor: (A: *) -> (X1 X2: ListAlg A) -> * :=
         (map: X1.X -> X2.X)
         (okNil: Eq (map X1.nil) (X2.nil))
         (okCons: ∀ (a: A) -> ∀ (x: X1) ->
                  Eq (map X1.cons a x) (X2.cons a (map x)))

Take a List Limit


  record ListLim: (A: *) -> * :=
         (point:  ∀ (x: ListAlg A) -> x.X)
         (limit:  ∀ (x1 x2: ListAlg A)
               -> ∀ (m: ListMor A x1 x2)
               -> Eq (map m point x1) (point x2))
